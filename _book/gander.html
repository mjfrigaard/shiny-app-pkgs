mod_counts_tbl_server <- function(id, vals) {
  moduleServer(id, function(input, output, session) {
    
    output$counts_table <- reactable::renderReactable({
      req(vals())

        thtr_rel_year >= vals()$start_year &
        thtr_rel_year <= vals()$end_year
